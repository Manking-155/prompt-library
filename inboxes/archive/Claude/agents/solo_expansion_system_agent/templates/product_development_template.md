## Development Standards

### Code Quality Standards
```dart
// Code example with best practices
class ProductController extends GetxController {
  // Reactive variables
  final _isLoading = false.obs;
  final _products = <Product>[].obs;
  final _error = Rx<String?>(null);

  // Getters
  bool get isLoading => _isLoading.value;
  List<Product> get products => _products;
  String? get error => _error.value;

  // Constructor with dependency injection
  ProductController({
    required GetProductsUseCase getProductsUseCase,
    required CreateProductUseCase createProductUseCase,
  })  : _getProductsUseCase = getProductsUseCase,
        _createProductUseCase = createProductUseCase;

  // Business logic methods
  Future<void> loadProducts() async {
    try {
      _isLoading.value = true;
      _error.value = null;

      final result = await _getProductsUseCase.execute();

      result.fold(
        (failure) => _error.value = failure.message,
        (products) => _products.assignAll(products),
      );
    } catch (e) {
      _error.value = 'Unexpected error occurred';
    } finally {
      _isLoading.value = false;
    }
  }
}
```

### Testing Strategy
```dart
// Unit test example
void main() {
  group('ProductController', () {
    late ProductController controller;
    late MockGetProductsUseCase mockGetProductsUseCase;

    setUp(() {
      mockGetProductsUseCase = MockGetProductsUseCase();
      controller = ProductController(
        getProductsUseCase: mockGetProductsUseCase,
        createProductUseCase: MockCreateProductUseCase(),
      );
    });

    test('should load products successfully', () async {
      // Arrange
      final mockProducts = [Product(id: '1', name: 'Test Product')];
      when(() => mockGetProductsUseCase.execute())
          .thenAnswer((_) async => Result.success(mockProducts));

      // Act
      await controller.loadProducts();

      // Assert
      expect(controller.products.length, 1);
      expect(controller.products.first.name, 'Test Product');
      expect(controller.isLoading, false);
      expect(controller.error, null);
    });
  });
}
```

class OnboardingController extends GetxController {
  final _variant = 'A'.obs; // 'A' or 'B'

  String get variant => _variant.value;

  void initializeVariant() {
    // Randomly assign variant to new users
    _variant.value = math.Random().nextBool() ? 'A' : 'B';

    // Track variant assignment
    AnalyticsService.trackEvent('onboarding_variant_assigned', {
      'variant': _variant.value,
      'user_id': userId,
    });
  }

  void completeOnboarding() {
    AnalyticsService.trackEvent('onboarding_completed', {
      'variant': _variant.value,
      'user_id': userId,
      'completion_time': DateTime.now().difference(_startTime).inSeconds,
    });
  }
}
